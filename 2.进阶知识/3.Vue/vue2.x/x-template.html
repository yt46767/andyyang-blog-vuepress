# x-template 使用

<script type="text/x-template" id="yourId">
<template>
  <h1 v-show="showHeading">
    <slot></slot>
  </h1>
  // ...
</template>
</script>

Vue.component('h1-heading', {
  template: '#yourId',
  props: {
    showHeading: {
      type: Boolean,
      default: true
    }
  }
})
